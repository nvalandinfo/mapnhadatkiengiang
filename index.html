function executeFilter() { // Lấy giá trị từ các bộ lọc const typeVal = document.getElementById('fType').value; const priceVal = document.getElementById('fPrice').value; const areaVal = document.getElementById('fArea').value; const directionVal = document.getElementById('fDirection').value; // Lọc dữ liệu chính xác const filtered = bdsData.filter(item => { // Kiểm tra Loại nhà đất const matchType = typeVal === "" || item.loai === typeVal; // Kiểm tra Mức giá (Giả định logic so sánh dựa trên thuộc tính gia hoặc giaSo của dữ liệu) // Lưu ý: Để lọc chính xác tuyệt đối, dữ liệu mẫu cần có các thuộc tính tương ứng const matchPrice = priceVal === "" || item.gia.includes(priceVal) || (item.mucGia === priceVal); // Kiểm tra Diện tích const matchArea = areaVal === "" || (item.dienTichPhanLoai === areaVal); // Kiểm tra Hướng const matchDirection = directionVal === "" || item.huong === directionVal; return matchType && matchPrice && matchArea && matchDirection; }); // Cập nhật bản đồ và danh sách renderData(filtered); // Đóng bảng lọc toggleFilterPanel(); // Tự động mở thanh danh sách để hiển thị kết quả document.getElementById('sideListing').classList.add('mobile-active'); }
